} } function doOptions(e) { return ContentService.createTextOutput("").setMimeType(ContentService.MimeType.TEXT); } /** * 🌟 新增：自動萃取總數的輔助函式 * 遇到 "0/6" 會自動抓出 6，遇到 "6" 還是 6 */ function extractTotal(val) { if (!val) return 0; if (typeof val === 'string' && val.includes('/')) { return parseInt(val.split('/')[1], 10) || 0; } return parseInt(val, 10) || 0; } /** * 包裹 3D 旋轉比對演算法：判斷包裹是否塞得進指定的格位 */ function boxCanFit(userDims, sizeStr) { // 將使用者輸入的長寬高由大到小排序 const p = userDims.slice().sort((a, b) => b - a); // 解析箱體尺寸 (例如 "340x270" -> 34cm x 27cm)，i 郵箱標準深度為 47cm const parts = sizeStr.split('x'); const boxW = parseInt(parts[0], 10) / 10; const boxH = parseInt(parts[1], 10) / 10; const boxD = 47; // 將箱體尺寸由大到小排序 const b = [boxW, boxH, boxD].sort((a, b) => b - a); // 每個對應的邊都必須小於或等於箱體 return p[0] <= b[0] && p[1] <= b[1] && p[2] <= b[2]; } /** * 核心搜尋功能 (雙重過濾混合戰術) */ function searchIBox(params) { const inputData = params.inputData; const dims = params.dims; // [L, W, H] 或 null const hasDims = (dims && dims.length === 3); const iBoxes = []; for (let i = 1; i < data.length; i++) { const lat = data[i][6]; const lng = data[i][7]; if (!lat || !lng) continue; // 🌟 修改：讀取試算表既有格位時，套用 extractTotal 自動過濾斜線 // 🛡️ 並且「刪除」 500x500 與 900x500 這兩個特殊/公務格位，不對一般民眾開放 let backupBoxes = { '340x270': extractTotal(data[i][10]), '140x170': extractTotal(data[i][11]), '340x170': extractTotal(data[i][12]), '340x240': extractTotal(data[i][15]), '340x310': extractTotal(data[i][16]), '340x120': extractTotal(data[i][17]), '180x220': extractTotal(data[i][18]), '340x200': extractTotal(data[i][19]), '340x110': extractTotal(data[i][20]) }; // 🌟 戰術第一關：硬體初篩 (如果輸入尺寸，就看機器有沒有做這個格子) if (hasDims) { let hardwareSupports = false; for (let size in backupBoxes) { if (backupBoxes[size] > 0 && boxCanFit(dims, size)) { hardwareSupports = true; break; } } if (!hardwareSupports) continue; // 機器根本裝不下，無情踢除！ } let updateTimeStr = ''; if (data[i][9]) { try { const d = new Date(data[i][9]); if (!isNaN(d.getTime())) updateTimeStr = Utilities.formatDate(d, "Asia/Taipei", "yyyy/MM/dd HH:mm"); else updateTimeStr = data[i][9].toString(); } catch(e) { updateTimeStr = data[i][9].toString(); } } iBoxes.push({ id: data[i][0].toString().trim(), name: data[i][2], address: data[i][3], locationDesc: data[i][4], updateTime: updateTimeStr, lat: lat, lng: lng, straightDist: Math.sqrt(Math.pow(lat - userCoord.lat, 2) + Math.pow(lng - userCoord.lng, 2)), boxes: backupBoxes }); } if (iBoxes.length === 0) throw new Error('附近沒有符合您包裹尺寸的 i 郵箱，請嘗試重新搜尋。'); // 🌟 擴大備胎：選出直線距離最近的 15 名 const candidates = iBoxes.sort((a, b) => a.straightDist - b.straightDist).slice(0, 15); // 🌟 獲取絕對即時的格位資料 fetchLiveBoxStatusById(candidates); // 🌟 戰術第二關：即時空位淘汰 let validCandidates = candidates; if (hasDims) { validCandidates = candidates.filter(cand => { let hasAvailableSlot = false; for (let size in cand.boxes) { let val = cand.boxes[size]; let remain = 0; if (typeof val === 'string' && val.includes('/')) { remain = parseInt(val.split('/')[1], 10) - parseInt(val.split('/')[0], 10); } else { remain = parseInt(val, 10); } // 裝得下，且目前剩餘大於 0 if (remain > 0 && boxCanFit(dims, size)) { hasAvailableSlot = true; break; } } return hasAvailableSlot; }); } if (validCandidates.length === 0) { throw new Error('附近的 i 郵箱「裝得下您包裹的格位」目前全部客滿，請縮小尺寸或稍後再試！'); } // 🌟 戰術第三關：只取前 10 個送給 Google 算真實導航 (保證不爆 Google Quota) const routingCandidates = validCandidates.slice(0, 10); // 🛡️ 過濾：即時抓取時，一樣遇到這兩種特殊大格子就直接跳過不處理 if (sizeKey === '500x500' || sizeKey === '900x500') return; const used = box.TotalCount - box.FreeCount; newBoxes[sizeKey] = used + '/' + box.TotalCount; }); cand.boxes = newBoxes;
